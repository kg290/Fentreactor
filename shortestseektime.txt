#include <stdio.h>
#include <stdlib.h>

#define MAX 200

int main(){
    int thm, disk_req[MAX], n, min,idx, pending, fin[MAX], head,diff;
    printf("Enter total disk Request: ");
    scanf("%d",&n);
    printf("Enter request sequence: \n");
    for(int i = 0 ; i<n ; i++){
        printf("%d: ",i+1);
        scanf("%d",&disk_req[i]);
        fin[i]=0;
    }
    printf("Enter Head: ");
    scanf("%d",&head);
    pending = n;
    printf("Disk scheduling sequence is : ");
    while(pending>0){
        min = 9999;
        for(int i = 0 ; i<n ; i++){
            if(fin[i]==0){
                diff = abs(head - disk_req[i]);
                if(diff<min){
                    min = diff;
                    idx = i;
                }
            }
        }
        fin[idx] = 1;
        thm+= abs(head - disk_req[idx]);
        pending--;
        head = disk_req[idx];
        printf("%d ",disk_req[idx]);
    }

    printf("\n");
    printf("Total head movement : %d",thm);
}
